148:         /**
149:     * Saves borrowing request. Returns operation status in JSON format.
150:         * Gets input via POST data.
151:         * 
152:         * @return JSON
153:     */
154:         public function doCheckout()
155:         {
156:             // Get post data
157:             $packets    = Input::get('packets');
158:             $borrowerId = Input::get('borrowerId');
159: 
160:             // Check if post data is an array (otherwise returns an empty array)
161:             $newPackets = array();
162:             if(is_array($packets) && isset($borrowerId)) 
163:             {
164:                 foreach($packets as $packet)
165:                 {
166:                     $databasePacket = Packet::find($packet['id']);
167:                     
168:                     // Checks if packet is registered in the database
169:                     if (!is_null($databasePacket)) 
170:                     {
171:                         // It exists, but may have already been reserved to another person
172:                         if (isset($databasePacket->borrower_id)) 
173:                         {
174:                             $packet['ok']     = false;
175:                             $packet['response'] = 'RESERVED';
176:                         } else
177:                         {
178:                             // Reserve packet for user
179:                             $databasePacket->borrower_id    = $borrowerId;
180:                             $databasePacket->requested_at   = Carbon::now();
181:                             $databasePacket->reserved_until = Carbon::now()->addWeek();
182:                             
183:                             if ($databasePacket->save())
184:                                 $packet['ok'] = true;
185:                             else {
186:                                 $packet['ok'] = false; // Error while saving
187:                                 $packet['response'] = 'FAILURE';
188:                             }
189:                         }
190:                     } else // Packet was not found
191:                     {
192:                         $packet['ok']     = false;
193:                         $packet['response'] = 'NULL';
194:                     }
195:                     
196:                     // Add packet to array
197:                     $newPackets[] = $packet;
198:                 }
199:             }
200:             
201:             return $newPackets;
202:         }
